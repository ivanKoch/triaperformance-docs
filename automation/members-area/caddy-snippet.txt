@membersPublicPages {
	path /members/login /members/forgot-password
	method GET
}
handle @membersPublicPages {
	root * /var/www/triaperformance/website
	file_server
}

@membersAuthPost {
	path /members/login /members/logout
	method POST
}
handle @membersAuthPost {
	reverse_proxy 127.0.0.1:8091
}

@membersForgotPost {
	path /members/forgot-password
	method POST
}
handle @membersForgotPost {
	rewrite * /webhook/resend-password
	reverse_proxy 100.70.89.17:5678
}

handle /members/* {
	forward_auth 127.0.0.1:8091 {
		uri /members/check
		copy_headers Cookie

		@denied status 401
		handle_response @denied {
			redir * /members/login?next={http.request.orig_uri} 302
		}
	}
	root * /var/www/triaperformance/website
	file_server
}
